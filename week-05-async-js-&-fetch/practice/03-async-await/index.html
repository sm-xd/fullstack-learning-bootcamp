Async/Await

Syntax Comparison

Promise .then()

fetchData()
  .then(data => {
    return process(data);
  })
  .then(result => {
    console.log(result);
  })
  .catch(err => {
    console.error(err);
  });

async/await

async function getData() {
  try {
    const data = await fetchData();
    const result = await process(data);
    console.log(result);
  } catch (err) {
    console.error(err);
  }
}
